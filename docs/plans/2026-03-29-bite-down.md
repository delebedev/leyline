**Key finding from code inspection:**

`GameEventCollector.visit(GameEventCardDamaged)` already captures `ev.source().id` as `sourceCardId`. `combatAnnotations()` in `AnnotationPipeline` already uses `idResolver(ev.sourceCardId)` as the `affectorId` in `AnnotationBuilder.damageDealt()`. The non-combat `GameEventCardDamaged` event fires during spell resolution. `combatAnnotations()` is called unconditionally from `computeAnnotations()` (called on every `buildFromGame`), gated only by whether the event queue contains `DamageDealtToCard` events. - [ ] **Step 1: Create `BiteDownTest.kt`**

```kotlin
package leyline.conformance

import io.kotest.assertions.assertSoftly
import io.kotest.core.spec.style.FunSpec
import io.kotest.matchers.booleans.shouldBeTrue
import io.kotest.matchers.collections.shouldContain
import io.kotest.matchers.nulls.shouldNotBeNull
import io.kotest.matchers.shouldBe
import leyline.IntegrationTag
import leyline.bridge.ForgeCardId
import leyline.bridge.SeatId
import wotc.mtgo.gre.external.messaging.Messages.AnnotationType
import forge.game.zone.ZoneType as ForgeZoneType

/**
 * Integration test for Bite Down (grpId 93925).
 *
 * Wire spec: docs/card-specs/bite-down.md
 * Design spec: docs/specs/2026-03-29-bite-down-damage-design.md
 *
 * Key assertion: DamageDealt.affectorId = dealing creature iid (NOT the spell iid).
 * The annotation pipeline uses GameEvent.DamageDealtToCard.sourceCardId which is
 * set from GameEventCardDamaged.source() â€” the creature that dealt damage.
 */
class BiteDownTest :
    FunSpec({

        tags(IntegrationTag)

        var harness: MatchFlowHarness? = null

        afterEach {
            harness?.shutdown()
            harness = null
        }

        test("Bite Down: DamageDealt affectorId is the dealing creature, not the spell") {
            val h = MatchFlowHarness(seed = 42L, validating = false)
            harness = h
            h.connectAndKeepPuzzle("puzzles/bite-down.pzl")

            // Resolve the dealing creature's instanceId before casting
            val human = h.game().registeredPlayers.first()
            val dealerCard = human.getZone(ForgeZoneType.Battlefield).cards
                .first { it.name == "Grizzly Bears" }
            val dealerIid = h.bridge.getOrAllocInstanceId(ForgeCardId(dealerCard.id)).value

            // Resolve the target creature's instanceId
            val ai = h.game().registeredPlayers.last()
            val targetCard = ai.getZone(ForgeZoneType.Battlefield).cards
                .first { it.name == "Grizzly Bears" }
            val targetIid = h.bridge.getOrAllocInstanceId(ForgeCardId(targetCard.id)).value

            // Cast Bite Down
            h.castSpellByName("Bite Down").shouldBeTrue()

            // Bite Down uses two-group SelectTargetsReq:
            //   Group 1 (targetIdx=1): your creature (the dealer)
            //   Group 2 (targetIdx=2): opponent's creature/PW (the target)
            // selectTargets sends both in one SelectTargetsResp, then SubmitTargetsReq.
            h.selectTargets(listOf(dealerIid, targetIid))

            // Capture the resolution diff
            val snap = h.messageSnapshot()
            h.passPriority()
            val resolutionMsgs = h.messagesSince(snap)

            // Find the GSM containing the DamageDealt annotation
            val damageGsm = resolutionMsgs
                .filter { it.hasGameStateMessage() }
                .map { it.gameStateMessage }
                .firstOrNull { gsm ->
                    gsm.annotationsList.any { AnnotationType.DamageDealt_af5a in it.typeList }
                }
            damageGsm.shouldNotBeNull()

            val damageAnn = damageGsm.annotation(AnnotationType.DamageDealt_af5a)

            assertSoftly {
                // Core assertion: affectorId is the dealing creature, not the spell
                damageAnn.affectorId shouldBe dealerIid

                // Target creature is in affectedIds
                damageAnn.affectedIdsList shouldContain targetIid

                // Damage amount = dealer power (Grizzly Bears = 2)
                damageAnn.detailInt("damage") shouldBe 2
            }

            // SBA_Damage fires in the same diff as resolution (target dies)
            val sbaZt = damageGsm.findZoneTransfer(
                // Target was reallocated â€” look for SBA_Damage category on any ZoneTransfer
                damageGsm.annotationsList
                    .filter { AnnotationType.ZoneTransfer_af5a in it.typeList }
                    .firstOrNull { ann -> ann.detailString("category") == "SBA_Damage" }
                    ?.affectedIdsList?.firstOrNull() ?: -1
            )
            sbaZt.shouldNotBeNull()
            sbaZt.category shouldBe "SBA_Damage"

            h.accumulator.assertConsistent("after Bite Down resolution")
        }

        test("Bite Down: spell resolves to graveyard with Resolve category") {
            val h = MatchFlowHarness(seed = 42L, validating = false)
            harness = h
            h.connectAndKeepPuzzle("puzzles/bite-down.pzl")

            val human = h.game().registeredPlayers.first()
            val ai = h.game().registeredPlayers.last()

            val dealerCard = human.getZone(ForgeZoneType.Battlefield).cards
                .first { it.name == "Grizzly Bears" }
            val dealerIid = h.bridge.getOrAllocInstanceId(ForgeCardId(dealerCard.id)).value

            val targetCard = ai.getZone(ForgeZoneType.Battlefield).cards
                .first { it.name == "Grizzly Bears" }
            val targetIid = h.bridge.getOrAllocInstanceId(ForgeCardId(targetCard.id)).value

            h.castSpellByName("Bite Down").shouldBeTrue()
            h.selectTargets(listOf(dealerIid, targetIid))

            val snap = h.messageSnapshot()
            h.passPriority()
            val msgs = h.messagesSince(snap)

            // Bite Down spell resolves to GY via "Resolve" category (Stackâ†’P1_GY)
            val resolveMsg = msgs.firstWithTransferCategory("Resolve")
            resolveMsg.shouldNotBeNull()

            // Spell is in human's GY after resolution
            val biteDownInGy = human.getZone(ForgeZoneType.Graveyard).cards
                .any { it.name == "Bite Down" }
            biteDownInGy.shouldBeTrue()
        }
    })
```
